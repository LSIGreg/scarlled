    // ---------- DATA MODELS ----------
    let db = {
        users: [],
        posts: [],
        stories: [],
        likes: [],      // { userId, postId }
        subscriptions: [], // { followerId, followingId }
        notifications: []  // { toUserId, fromUserName, postId, message, read }
    };

    let currentUserId = null;
    let pendingSendPostId = null;

    // Load or init localStorage
    function loadData() {
        const stored = localStorage.getItem("social_db");
        if (stored) {
            db = JSON.parse(stored);
        } else {
            // demo data
            const demoUsers = [

            ];
            const demoPosts = [
                { id: 1, userId: 1, media: "https://picsum.photos/id/1015/600/400", mediaType: "image", caption: "Красивый пейзаж! 🏔️", createdAt: Date.now() - 86400000 },
                { id: 2, userId: 2, media: "https://www.w3schools.com/html/mov_bbb.mp4", mediaType: "video", caption: "Смешное видео 😂", createdAt: Date.now() - 172800000 },

            ];
            const demoLikes = [
                { userId: 2, postId: 1 },
                { userId: 3, postId: 1 }
            ];
            const demoStories = [
                { id: 1, userId: 1, media: "https://picsum.photos/id/101/300/300", mediaType: "image", createdAt: Date.now() - 10000000 },
                { id: 2, userId: 2, media: "https://picsum.photos/id/106/300/300", mediaType: "image", createdAt: Date.now() - 5000000 }
            ];
            db.users = demoUsers;
            db.posts = demoPosts;
            db.likes = demoLikes;
            db.stories = demoStories;
            db.subscriptions = [
                { followerId: 1, followingId: 2 },
                { followerId: 1, followingId: 3 },
                { followerId: 2, followingId: 1 }
            ];
            db.notifications = [];
            saveData();
        }
    }

    function saveData() {
        localStorage.setItem("social_db", JSON.stringify(db));
    }

    // Helper functions
    function getUserById(id) {
        return db.users.find(u => u.id === id);
    }

    function isFollowing(followerId, followingId) {
        return db.subscriptions.some(s => s.followerId === followerId && s.followingId === followingId);
    }

    function toggleFollow(followerId, followingId) {
        if (isFollowing(followerId, followingId)) {
            db.subscriptions = db.subscriptions.filter(s => !(s.followerId === followerId && s.followingId === followingId));
        } else {
            db.subscriptions.push({ followerId, followingId });
        }
        saveData();
    }

    function likePost(userId, postId) {
        const exists = db.likes.some(l => l.userId === userId && l.postId === postId);
        if (exists) {
            db.likes = db.likes.filter(l => !(l.userId === userId && l.postId === postId));
        } else {
            db.likes.push({ userId, postId });
        }
        saveData();
    }

    function isLiked(userId, postId) {
        return db.likes.some(l => l.userId === userId && l.postId === postId);
    }

    function getPostLikesCount(postId) {
        return db.likes.filter(l => l.postId === postId).length;
    }

    function addPost(userId, mediaData, mediaType, caption) {
        const newId = Date.now();
        db.posts.unshift({ id: newId, userId, media: mediaData, mediaType, caption, createdAt: Date.now() });
        saveData();
        renderFeed();
    }

    function addStory(userId, mediaData, mediaType) {
        const newId = Date.now();
        db.stories.unshift({ id: newId, userId, media: mediaData, mediaType, createdAt: Date.now() });
        saveData();
        renderStories();
    }

    function sendPostNotification(toUserId, fromUserName, postId) {
        db.notifications.push({ toUserId, fromUserName, postId, message: `отправил(а) вам публикацию`, read: false, createdAt: Date.now() });
        saveData();
        alert(`Пост отправлен пользователю ${getUserById(toUserId)?.name}`);
    }

    // Render functions
    function renderStories() {
        const container = document.getElementById("storiesContainer");
        if (!container) return;
        const currentUser = getUserById(currentUserId);
        let html = `<div class="story-item" id="addStoryBtn">
                        <div class="story-img story-add">+</div>
                        <div class="story-name">Добавить</div>
                    </div>`;
        db.stories.forEach(story => {
            const user = getUserById(story.userId);
            if (user) {
                html += `<div class="story-item" data-story-id="${story.id}" data-story-media="${story.media}" data-story-type="${story.mediaType}">
                            <img class="story-img" src="${story.media}" alt="story">
                            <div class="story-name">${user.name.split(' ')[0]}</div>
                         </div>`;
            }
        });
        container.innerHTML = html;
        document.querySelectorAll('.story-item[data-story-id]').forEach(el => {
            el.addEventListener('click', (e) => {
                e.stopPropagation();
                const media = el.dataset.storyMedia;
                const type = el.dataset.storyType;
                showStoryViewer(media, type);
            });
        });
        document.getElementById("addStoryBtn")?.addEventListener("click", () => {
            document.getElementById("storyModal").style.display = "flex";
        });
    }

    function showStoryViewer(media, type) {
        const viewer = document.getElementById("storyViewer");
        const img = document.getElementById("viewerImage");
        const video = document.getElementById("viewerVideo");
        img.style.display = "none";
        video.style.display = "none";
        if (type === "image") {
            img.src = media;
            img.style.display = "block";
        } else {
            video.src = media;
            video.style.display = "block";
            video.load();
        }
        viewer.style.display = "flex";
    }

    function renderFeed() {
        const container = document.getElementById("feedContainer");
        if (!container) return;
        // get posts from people current user follows + own posts
        const followingIds = db.subscriptions.filter(s => s.followerId === currentUserId).map(s => s.followingId);
        followingIds.push(currentUserId);
        const filteredPosts = db.posts.filter(p => followingIds.includes(p.userId));
        filteredPosts.sort((a,b) => b.createdAt - a.createdAt);
        
        if (filteredPosts.length === 0) {
            container.innerHTML = `<div style="background:white; padding:20px; text-align:center; border-radius:12px;">Нет постов. Подпишитесь на друзей или создайте свой первый пост!</div>`;
            return;
        }
        
        let html = "";
        filteredPosts.forEach(post => {
            const author = getUserById(post.userId);
            const liked = isLiked(currentUserId, post.id);
            const likesCount = getPostLikesCount(post.id);
            html += `
                <div class="post-card" data-post-id="${post.id}">
                    <div class="post-header">
                        <img class="post-avatar" src="${author?.avatar || 'https://via.placeholder.com/40'}" alt="avatar">
                        <div>
                            <div class="post-author">${author?.name}</div>
                            <div class="post-time">${new Date(post.createdAt).toLocaleString()}</div>
                        </div>
                    </div>
                    <div class="post-media-container">
                        ${post.mediaType === 'image' ? `<img class="post-media" src="${post.media}" alt="post image">` : `<video class="post-media" controls src="${post.media}"></video>`}
                    </div>
                    <div class="post-caption">${post.caption || ''}</div>
                    <div class="post-actions">
                        <button class="action-btn like-btn ${liked ? 'liked' : ''}" data-post-id="${post.id}">❤️ ${likesCount} ${liked ? 'Нравится' : 'Мне нравится'}</button>
                        <button class="action-btn send-btn" data-post-id="${post.id}">📤 Отправить</button>
                    </div>
                </div>
            `;
        });
        container.innerHTML = html;
        // attach like handlers
        document.querySelectorAll('.like-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const postId = parseInt(btn.dataset.postId);
                likePost(currentUserId, postId);
                renderFeed();
            });
        });
        document.querySelectorAll('.send-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                pendingSendPostId = parseInt(btn.dataset.postId);
                openSendModal();
            });
        });
    }

    function renderSuggestions() {
        const container = document.getElementById("suggestionsList");
        if (!container) return;
        const otherUsers = db.users.filter(u => u.id !== currentUserId);
        let html = "";
        otherUsers.forEach(user => {
            const following = isFollowing(currentUserId, user.id);
            html += `
                <div class="suggestion-user">
                    <div class="suggestion-info">
                        <img class="suggestion-avatar" src="${user.avatar || 'https://via.placeholder.com/36'}" alt="">
                        <span>${user.name}</span>
                    </div>
                    <button class="subscribe-btn ${following ? 'unsubscribe-btn' : ''}" data-user-id="${user.id}">${following ? 'Отписаться' : 'Подписаться'}</button>
                </div>
            `;
        });
        container.innerHTML = html;
        document.querySelectorAll('.subscribe-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const targetId = parseInt(btn.dataset.userId);
                toggleFollow(currentUserId, targetId);
                renderSuggestions();
                renderFeed();
            });
        });
    }

    function openSendModal() {
        const select = document.getElementById("sendToUser");
        const users = db.users.filter(u => u.id !== currentUserId);
        select.innerHTML = users.map(u => `<option value="${u.id}">${u.name}</option>`).join('');
        document.getElementById("sendModal").style.display = "flex";
    }

    // Authentication
    function register(name, username, email, password, avatarUrl) {
        if (db.users.some(u => u.email === email || u.username === username)) {
            alert("Пользователь с таким email или именем уже существует");
            return false;
        }
        const newId = Date.now();
        const avatar = avatarUrl || `https://randomuser.me/api/portraits/${Math.random() > 0.5 ? 'men' : 'women'}/${Math.floor(Math.random()*50)}.jpg`;
        db.users.push({ id: newId, name, username, email, password, avatar });
        saveData();
        alert("Регистрация успешна! Теперь войдите.");
        return true;
    }

    function login(emailOrUsername, password) {
        const user = db.users.find(u => (u.email === emailOrUsername || u.username === emailOrUsername) && u.password === password);
        if (user) {
            currentUserId = user.id;
            localStorage.setItem("currentUserId", currentUserId);
            showApp();
            return true;
        } else {
            alert("Неверные учетные данные");
            return false;
        }
    }

    function logout() {
        currentUserId = null;
        localStorage.removeItem("currentUserId");
        document.getElementById("appContainer").style.display = "none";
        document.getElementById("authContainer").style.display = "flex";
    }

    function showApp() {
        document.getElementById("authContainer").style.display = "none";
        document.getElementById("appContainer").style.display = "block";
        const user = getUserById(currentUserId);
        document.getElementById("currentUserName").innerText = user?.name;
        document.getElementById("currentUserAvatar").src = user?.avatar || 'https://via.placeholder.com/40';
        renderStories();
        renderFeed();
        renderSuggestions();
    }

    // UI events
    document.getElementById("doLogin")?.addEventListener("click", () => {
        const email = document.getElementById("loginEmail").value;
        const pwd = document.getElementById("loginPassword").value;
        login(email, pwd);
    });
    document.getElementById("showRegister")?.addEventListener("click", () => {
        document.getElementById("loginForm").style.display = "none";
        document.getElementById("registerForm").style.display = "block";
    });
    document.getElementById("showLogin")?.addEventListener("click", () => {
        document.getElementById("registerForm").style.display = "none";
        document.getElementById("loginForm").style.display = "block";
    });
    document.getElementById("doRegister")?.addEventListener("click", () => {
        const name = document.getElementById("regName").value;
        const username = document.getElementById("regUsername").value;
        const email = document.getElementById("regEmail").value;
        const pwd = document.getElementById("regPassword").value;
        const avatar = document.getElementById("regAvatar").value;
        if (!name || !username || !email || !pwd) {
            alert("Заполните все поля");
            return;
        }
        if (register(name, username, email, pwd, avatar)) {
            document.getElementById("registerForm").style.display = "none";
            document.getElementById("loginForm").style.display = "block";
        }
    });
    document.getElementById("logoutBtn")?.addEventListener("click", logout);

    const fab = document.getElementById("fab");
    const postModal = document.getElementById("postModal");
    const storyModal = document.getElementById("storyModal");
    const sendModal = document.getElementById("sendModal");
    fab?.addEventListener("click", () => { postModal.style.display = "flex"; });
    document.getElementById("closePostModal")?.addEventListener("click", () => { postModal.style.display = "none"; });
    document.getElementById("closeStoryModal")?.addEventListener("click", () => { storyModal.style.display = "none"; });
    document.getElementById("closeSendModal")?.addEventListener("click", () => { sendModal.style.display = "none"; });
    
    document.getElementById("submitPost")?.addEventListener("click", () => {
        const caption = document.getElementById("postCaption").value;
        const file = document.getElementById("postFile").files[0];
        if (!file) { alert("Выберите фото или видео"); return; }
        const type = file.type.startsWith("image/") ? "image" : "video";
        const reader = new FileReader();
        reader.onload = function(e) {
            addPost(currentUserId, e.target.result, type, caption);
            postModal.style.display = "none";
            document.getElementById("postCaption").value = "";
            document.getElementById("postFile").value = "";
        };
        reader.readAsDataURL(file);
    });
    
    document.getElementById("submitStory")?.addEventListener("click", () => {
        const file = document.getElementById("storyFile").files[0];
        if (!file) { alert("Выберите медиа для истории"); return; }
        const type = file.type.startsWith("image/") ? "image" : "video";
        const reader = new FileReader();
        reader.onload = function(e) {
            addStory(currentUserId, e.target.result, type);
            storyModal.style.display = "none";
            document.getElementById("storyFile").value = "";
        };
        reader.readAsDataURL(file);
    });
    
    document.getElementById("confirmSend")?.addEventListener("click", () => {
        const toUserId = parseInt(document.getElementById("sendToUser").value);
        const fromUser = getUserById(currentUserId);
        sendPostNotification(toUserId, fromUser.name, pendingSendPostId);
        sendModal.style.display = "none";

    });

        currentUserId = parseInt(savedUserId);
        showApp();

        document.getElementById("authContainer").style.display = "flex";
